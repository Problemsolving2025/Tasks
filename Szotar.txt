using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

using System.Data;
using System.Data.SqlClient;
using System.Web.Configuration;

namespace WebApplication1
{
    public partial class Szotar : System.Web.UI.Page
    {
        SqlConnection con = new SqlConnection(WebConfigurationManager.ConnectionStrings["adatbazis"].ConnectionString);

        protected void Page_Load(object sender, EventArgs e)
        {

        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            if(DropDownList1.SelectedIndex == 0){
                string szo = TextBox3.Text;

                try
                { 
                con.Open();

                SqlCommand cmd = new SqlCommand("select Szerb, Leiras from szavak where Angol = @angol", con);
                cmd.Parameters.AddWithValue("@angol", TextBox3.Text);

                SqlDataReader dr = cmd.ExecuteReader();
                dr.Read();
                if (dr.HasRows)
                {
                    TextBox1.Text = dr[1].ToString();
                    TextBox2.Text = dr[0].ToString();
                }
                dr.Close();
                
                }
                catch (Exception err)
                {

                    Response.Write(err.ToString());
                }
                finally
                {
                    con.Close();
                    
                }
            
            }
            else if (DropDownList1.SelectedIndex == 1)
            {
                string szo = TextBox3.Text;

                try
                {
                    con.Open();

                    SqlCommand cmd = new SqlCommand("select Angol, Leiras from szavak where szerb = @szerb", con);
                    cmd.Parameters.AddWithValue("@szerb", TextBox2.Text);

                    SqlDataReader dr = cmd.ExecuteReader();
                    dr.Read();
                    TextBox1.Text = dr[1].ToString();
                    TextBox3.Text = dr[0].ToString();

                    dr.Close();

                }
                catch (Exception err)
                {

                    Response.Write(err.ToString());
                }
                finally
                {
                    con.Close();

                }

            }
            else if (DropDownList1.SelectedIndex == 2)
            {
                Response.Write("<script> alert('Valassz ki egyet')</script>");
            
            }
        }
    }
}
