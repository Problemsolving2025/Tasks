using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

using System.Data;
using System.Data.SqlClient;
using System.Web.Configuration;
using System.Web.UI.DataVisualization.Charting;

namespace B6
{
    public partial class Grafika : System.Web.UI.Page
    {
        SqlConnection con = new SqlConnection(WebConfigurationManager.ConnectionStrings["adatbazis"].ConnectionString);

        protected void Page_Load(object sender, EventArgs e)
        {
            try {
                SqlCommand cmd = new SqlCommand("select Osztaly, AtlagJegy from Siker", con);

                DataTable dt = new DataTable();
                SqlDataAdapter da = new SqlDataAdapter(cmd);
                da.Fill(dt);
                //tombok letrehozasa
                string[] x = new string[dt.Rows.Count];
                decimal[] y = new decimal[dt.Rows.Count];
                //tombok feltoltese
                for (int i = 0; i < dt.Rows.Count; i++)
                {
                    x[i]=dt.Rows[i][0].ToString();
                    y[i]=Convert.ToDecimal(dt.Rows[i][1]);
                }

                // rajz
                Chart1.Series[0].Points.DataBindXY(x, y);
                Chart1.Series[0].ChartType = SeriesChartType.Column;

                //webconfig publish utan torolni a ChartImageHandler alatt a dir=c:\\TempImagesFiles\;
            }
            catch(Exception x){
                Response.Write(x.ToString());
            }
        }
    }
}
